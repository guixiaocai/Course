int main()
{
	int k,m,s;
	printf("please input k and m:\n");
	scanf("%d, %d",&k,&m);
	s = fibo(k,m);
	printf("the result is %d\n",s);
	return 0;
	
}

int fibo(int k, int m)
{
	int a[m+1];
	int i;
	for (i = 0; i < k - 1; i++)
	{
		a[i] = 0;
	}
	a[k-1] = a[k] = 1;

	for (i = k + 1; i < m + 1; i++)
	{
		a[i] = 2 * a[i-1] - a[i-k-1];
	}
	return a[m];
}

#include<stdio.h>
typedef enum {A, B, C, D, E} SchoolName;

typedef enum {Female, Male} Gender;

typedef struct
{
	char event[3];
	Gender sex;
	SchoolName school;
	int score;
} Info;

typedef struct
{
	int MaleScore;
	int FemaleScore;
	int TotalScore;
} Score;

int count(SchoolName s, Info a[], int n)
{
	Score temp;
	temp.MaleScore = 0;
	temp.FemaleScore = 0;
	temp.TotalScore = 0;
	int i;
	for(i = 0; i < n; i++)
	{
		if(a[i].school == s)
		{
			if(a[i].sex == Male)
				temp.MaleScore += a[i].score;
			if(a[i].sex == Female)
				temp.FemaleScore += a[i].score;
		}
	}
	temp.TotalScore = temp.MaleScore + temp.FemaleScore;
	return temp;
}

void poly(int i, int n, float a[], float x)
{
	if(i > 0)
		return a[n-i] + poly(i - 1, n, a, x) * x;
	else
		return a[n];
}
